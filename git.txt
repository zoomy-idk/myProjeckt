git init - для инициализации репозитория
git status - посмотреть статус

{
git config --global user.name "имя" 
git config --global user.email "имя"
} - чтобы было видно кто исправлял файлы

git add * - подготавливает изменения к сохранению
git commit -m "комментарий" - закомитить файл в репозиторий и добавить комментарий
git push - закинуть на GitHub
git clone - клонировать репозиторий
git pull - подтянуть обновленные данные с GitHub

{
ssh-keygen - создание ssh ключа для подключения к GitHub
}

git branch (name branch) - создать новую ветку
git checkout (name branch) - перейти на другую ветку
git branch -d (name branch) - удалить ветку
git checkout -b (name branch) - создать ветку, и сразу в нее перейти

git merge (name branch) -  слияние веток

git commit --amend - зайти в редактор, чтобы изменить коммит
git branch -m - переиминовать ветку
git push origin --delete (имя ветки) - удалить ветку на GitGub не переходя в него
